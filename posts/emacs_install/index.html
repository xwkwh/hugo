clone source code 

  git clone https://git.savannah.gnu.org/git/emacs.git
  cd emacs
  // checkout version
  git tag
  git checkout emacs-27.1

build & install 

  ./autogen.sh
  ./configure --without-selinux

  make -j4 && sudo make install
